 25def readCachedFile(key, secret, params, hash:tuple[str, str]=None):
 26    """pull cached/cache api results for parsing
 27
 28    Args:
 29        params (json): parameters for search, should not cause issue if 
 30        filled with more information than strictly necessary, given 
 31        parameter cleaning.
 32        hash (tuple[str, str], optional): (name of api, results). 
 33        Defaults to None. If none is entered, new results will be pulled and hashed.
 34
 35    Returns:
 36        json: results for covered and service reviews with format: 
 37        {
 38            "{hash[0]}": {
 39                "hash": "{hash{1}}",
 40                "data": "{api results}"
 41                }
 42        }
 43    """
 44    try:
 45        res = {}
 46        with open(f"..\\cache\\{hash[1]}", 'r', encoding='utf-8') as file:
 47            res["hash"] =  hash[1]
 48            res["type"] = hash[0]
 49            res["data"] = json.load(file)
 50        return res
 51    except Exception:
 52        _hash = retrieveInfo(key, secret, params, hash[0])
 53        return readCachedFile(key, secret, params, (hash[0], _hash))

 76def validateInsurance(key, secret, params, hash=None):
 77    """Pulls insurance information for all plans returned from params. 
 78
 79    Args:
 80        params (set of all search parameters)
 81        hash (string, optional): the hash for the coverage results if had. 
 82        Defaults to None. Pulls new info if none specified. 
 83
 84    Returns:
 85        list[json]: list of plan info, each plan result contains:
 86            status (active, etc.), statusCode, eligibilityStartDate, eligibilityEndDate,
 87            insuranceTypeCode (HMO?, etc), type (Medical Care, Vision, etc.)
 88    """
 89
 90    if hash is None:
 91        coverages = getFromApi(key, secret, params, "coverages")['coverages'][0]
 92    else:
 93        ApiResults = readCachedFile(key, secret, params, ("coverages", hash))
 94        coverages = ApiResults['data']['coverages'][0]
 95
 96    res = []
 97
 98    # There should only be 1-3 plans, if theres more than that narrow search parameters. 
 99    for plan in coverages['plans']:
100        planInfo = {}
101        for item in ['name','status','statusCode','eligibilityStartDate','eligibilityEndDate',
102                     'insuranceTypeCode','type']:
103            try:
104                planInfo[item] = plan[item]
105            except:
106                continue
107
108        res.append(planInfo)
109
110    res.append({'OutOfState': coverages['supplementalInformation']['outOfArea']})
111    return res

194def patientPaymentInfo(key, secret, params, hash=None):
195    """Get all payment/benefit info (deductibles, coPay, etc.) for Medical Care benefits
196
197    Args:
198        params (json): all search parameters
199        hash (string, optional): the hash for the coverage results if had. 
200        Defaults to None. Pulls new info if none specified. 
201
202    Returns:
203        list[json]: list of plans and all of their benefits
204    """
205    #TODO: figure out how to determine if in network
206    network = 'inNetwork'
207
208    coverages = getFromApi(key, secret, params, "coverages")['coverages'][0]
209    # ApiResults = readCachedFile(key, secret, params, ("coverages",hash))
210    # coverages = ApiResults['data']['coverages'][0]
211
212    res = [] 
213    for plan in coverages['plans']:
214        planBen = {"status":plan["status"]}
215        for benefit in plan['benefits']:
216            if benefit['name'] != 'Medical Care':
217                continue
218
219            benefits = {}
220            try: 
221                benefits["OOPRemaining"] = benefit["CostContainment"]   # cost containment is OOP remaining
222            except:
223                pass
224
225            for i in ['outOfPocket', 'deductibles', 'coInsurance', 'coPayment']:
226                benefits[i] = None
227
228                try:
229                    benefits[i] = (benefit['amounts'][i][network]['amount'],
230                                benefit['amounts'][i][network]['unit'])
231                except:
232                    continue
233
234            planBen[benefit["name"]] = benefits
235            
236        res.append(planBen)
237
238    return res
